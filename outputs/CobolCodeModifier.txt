      *----------------------------------------------------------------*

           05  WS-PROGRAM-NAME     PIC X(08) VALUE 'PORTUPDT '.

           05  WS-ERROR            PIC S9(4) VALUE +8.
           
       01  WS-SWITCHES.
           05  WS-FILE-STATUS      PIC X(02).
               88  WS-SUCCESS-STATUS     VALUE '00'.
               88  WS-EOF-STATUS        VALUE '10'.
               88  WS-REC-NOT-FND       VALUE '23'.
           
           05  WS-UPDT-STATUS      PIC X(02).
               88  WS-UPDT-SUCCESS      VALUE '00'.
               88  WS-UPDT-EOF          VALUE '10'.
      *-- Change: Add price feed file status
           05  WS-PRC-STATUS       PIC X(02).

               88  WS-PRC-EOF           VALUE '10'.
           
           05  WS-END-OF-FILE-SW   PIC X     VALUE 'N'.
               88  END-OF-FILE              VALUE 'Y'.
               88  NOT-END-OF-FILE          VALUE 'N'.
           
      *----------------------------------------------------------------*
      * Work areas
      *----------------------------------------------------------------*
       01  WS-WORK-AREAS.
           05  WS-UPDATE-COUNT     PIC 9(7) VALUE ZERO.
           05  WS-ERROR-COUNT      PIC 9(7) VALUE ZERO.
           05  WS-RETURN-CODE      PIC S9(4) VALUE +0.
           05  WS-NUMERIC-WORK     PIC S9(13)V99.
      *-- Change: Add real-time price and timestamp
           05  WS-REALTIME-PRICE   PIC S9(13)V99 COMP-3.
           05  WS-REALTIME-TIMESTAMP PIC 9(14).
      *-- Change: Audit log work area
           05  WS-AUDIT-MESSAGE    PIC X(80).
       
       PROCEDURE DIVISION.
       0000-MAIN.
           PERFORM 1000-INITIALIZE
           
           PERFORM 1005-LOAD-REALTIME-PRICE-FEED *-- Change: Load price feed
           
           PERFORM 2000-PROCESS
              UNTIL END-OF-FILE
           
           PERFORM 3000-TERMINATE
           
           GOBACK.
           
       1000-INITIALIZE.
           INITIALIZE WS-WORK-AREAS
           
           OPEN I-O   PORTFOLIO-FILE
           OPEN INPUT UPDATE-FILE
      *-- Change: Open price feed file
           OPEN INPUT PRICE-FEED-FILE
           
           IF NOT WS-SUCCESS-STATUS OR 
              NOT WS-UPDT-SUCCESS OR
              NOT WS-PRC-SUCCESS
              DISPLAY 'Error opening files: ' 
                      'PORT=' WS-FILE-STATUS
                      'UPDT=' WS-UPDT-STATUS
                      'PRC='  WS-PRC-STATUS
              MOVE WS-ERROR TO WS-RETURN-CODE
              PERFORM 3000-TERMINATE
           END-IF
           .
      *-- Change: Load first price feed record
       1005-LOAD-REALTIME-PRICE-FEED.
           READ PRICE-FEED-FILE
               AT END
                   MOVE ZERO TO WS-REALTIME-PRICE
                   MOVE ZERO TO WS-REALTIME-TIMESTAMP
               NOT AT END
                   MOVE PRCFEED-PRICE TO WS-REALTIME-PRICE
                   MOVE PRCFEED-TIMESTAMP TO WS-REALTIME-TIMESTAMP
           END-READ
           .
           
       2000-PROCESS.
           READ UPDATE-FILE
               AT END
                   SET END-OF-FILE TO TRUE
               NOT AT END
                   PERFORM 2100-PROCESS-UPDATE
           END-READ
           .
           
       2100-PROCESS-UPDATE.
           MOVE UPDT-KEY TO PORT-KEY
           
           READ PORTFOLIO-FILE
           
           IF WS-SUCCESS-STATUS
               PERFORM 2200-APPLY-UPDATE
           ELSE
               ADD 1 TO WS-ERROR-COUNT
               DISPLAY 'Record not found: ' PORT-KEY
      *-- Change: Log error to audit
               MOVE 'Record not found: ' TO WS-AUDIT-MESSAGE
               PERFORM 2500-AUDIT-LOGGING
           END-IF
           .
           
       2200-APPLY-UPDATE.
           EVALUATE TRUE
               WHEN UPDT-STATUS
                   MOVE UPDT-NEW-VALUE TO PORT-STATUS
               WHEN UPDT-NAME
                   MOVE UPDT-NEW-VALUE TO PORT-CLIENT-NAME
               WHEN UPDT-VALUE
                   MOVE UPDT-NEW-VALUE TO WS-NUMERIC-WORK
                   MOVE WS-NUMERIC-WORK TO PORT-TOTAL-VALUE
           END-EVALUATE
      *-- Change: Apply real-time price update and recalculate valuation
           IF WS-REALTIME-PRICE > ZERO
               MOVE WS-REALTIME-PRICE TO PORT-TOTAL-VALUE
               MOVE WS-REALTIME-TIMESTAMP TO PORT-LAST-MAINT
               MOVE 'Real-time price update applied' TO WS-AUDIT-MESSAGE
               PERFORM 2500-AUDIT-LOGGING
           END-IF
           
           REWRITE PORT-RECORD
           
           IF WS-SUCCESS-STATUS
               ADD 1 TO WS-UPDATE-COUNT
           ELSE
               ADD 1 TO WS-ERROR-COUNT
               DISPLAY 'Update failed for: ' PORT-KEY
      *-- Change: Log error to audit
               MOVE 'Update failed for: ' TO WS-AUDIT-MESSAGE
               PERFORM 2500-AUDIT-LOGGING
           END-IF
           .
      *-- Change: Audit logging paragraph
       2500-AUDIT-LOGGING.
           DISPLAY 'AUDIT: ' WS-AUDIT-MESSAGE
           .
           
       3000-TERMINATE.
           CLOSE PORTFOLIO-FILE
                 UPDATE-FILE
                 PRICE-FEED-FILE
           
           DISPLAY 'Updates processed: ' WS-UPDATE-COUNT
           DISPLAY 'Errors occurred:  ' WS-ERROR-COUNT
           
           MOVE WS-RETURN-CODE TO RETURN-CODE
           .
