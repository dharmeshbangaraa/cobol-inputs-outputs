      *================================================================*
      * Program Name: PORTUPDT
      * Description: Portfolio Update Program
      *             Updates existing portfolio records
      * Author: [Author name]
      * Date Written: 2024-03-20
      * Maintenance Log:
      * Date       Author        Description
      * ---------- ------------- -------------------------------------
      * 2024-03-20 [Author]     Initial Creation
      * 2024-06-XX [COBOL Impact Modifier Agent] Real-time price feed integration *-- Change: Real-time price feed logic and error handling added
      *================================================================*
       IDENTIFICATION DIVISION.
       PROGRAM-ID. PORTUPDT.
       
       ENVIRONMENT DIVISION.
       CONFIGURATION SECTION.
       SOURCE-COMPUTER. IBM-ZOS.
       OBJECT-COMPUTER. IBM-ZOS.
       
       INPUT-OUTPUT SECTION.
       FILE-CONTROL.
           SELECT PORTFOLIO-FILE
               ASSIGN TO PORTFILE
               ORGANIZATION IS INDEXED
               ACCESS MODE IS RANDOM
               RECORD KEY IS PORT-KEY
               FILE STATUS IS WS-FILE-STATUS.
           
           SELECT UPDATE-FILE
               ASSIGN TO UPDTFILE
               ORGANIZATION IS SEQUENTIAL
               FILE STATUS IS WS-UPDT-STATUS.

      *-- Change: Add real-time price feed file definition
           SELECT PRICEFEED-FILE
               ASSIGN TO PRICEFEED
               ORGANIZATION IS SEQUENTIAL
               FILE STATUS IS WS-PRICEFEED-STATUS.
       
       DATA DIVISION.
       FILE SECTION.
       FD  PORTFOLIO-FILE.
           COPY PORTFLIO.
           
       FD  UPDATE-FILE.
       01  UPDATE-RECORD.
           05  UPDT-KEY.
               10  UPDT-ID        PIC X(8).
               10  UPDT-ACCT-NO   PIC X(10).
           05  UPDT-ACTION        PIC X(1).
               88  UPDT-STATUS    VALUE 'S'.
               88  UPDT-VALUE     VALUE 'V'.
               88  UPDT-NAME      VALUE 'N'.
           05  UPDT-NEW-VALUE     PIC X(50).

      *-- Change: Add price feed record definition
       FD  PRICEFEED-FILE.
       01  PRICEFEED-RECORD.
           05  PF-PORTFOLIO-ID    PIC X(8).
           05  PF-PRICE           PIC S9(13)V99 COMP-3.
           05  PF-TIMESTAMP       PIC 9(14).

       WORKING-STORAGE SECTION.
      *----------------------------------------------------------------*
      * Constants and switches
      *----------------------------------------------------------------*
       01  WS-CONSTANTS.
           05  WS-PROGRAM-NAME     PIC X(08) VALUE 'PORTUPDT '.
           05  WS-SUCCESS          PIC S9(4) VALUE +0.
           05  WS-ERROR            PIC S9(4) VALUE +8.
           
       01  WS-SWITCHES.
           05  WS-FILE-STATUS      PIC X(02).
               88  WS-SUCCESS-STATUS     VALUE '00'.
               88  WS-EOF-STATUS        VALUE '10'.
               88  WS-REC-NOT-FND       VALUE '23'.
           
           05  WS-UPDT-STATUS      PIC X(02).
               88  WS-UPDT-SUCCESS      VALUE '00'.
               88  WS-UPDT-EOF          VALUE '10'.

      *-- Change: Add price feed file status
           05  WS-PRICEFEED-STATUS PIC X(02).
               88  WS-PRICEFEED-SUCCESS VALUE '00'.
               88  WS-PRICEFEED-EOF     VALUE '10'.
           
           05  WS-END-OF-FILE-SW   PIC X     VALUE 'N'.
               88  END-OF-FILE              VALUE 'Y'.
               88  NOT-END-OF-FILE          VALUE 'N'.
           
      *----------------------------------------------------------------*
      * Work areas
      *----------------------------------------------------------------*
       01  WS-WORK-AREAS.
           05  WS-UPDATE-COUNT     PIC 9(7) VALUE ZERO.
           05  WS-ERROR-COUNT      PIC 9(7) VALUE ZERO.
           05  WS-RETURN-CODE      PIC S9(4) VALUE +0.
           05  WS-NUMERIC-WORK     PIC S9(13)V99.

      *-- Change: Add work areas for price feed and error handling
           05  WS-PRICE-FOUND      PIC X VALUE 'N'.
           05  WS-FEED-ERROR       PIC X VALUE 'N'.
           05  WS-FEED-ERROR-CODE  PIC X(04).
           05  WS-FEED-TIMESTAMP   PIC 9(14).
           05  WS-FEED-PRICE       PIC S9(13)V99.

       PROCEDURE DIVISION.
       0000-MAIN.
           PERFORM 1000-INITIALIZE
           
           PERFORM 2000-PROCESS
              UNTIL END-OF-FILE
           
           PERFORM 3000-TERMINATE
           
           GOBACK.
           
       1000-INITIALIZE.
           INITIALIZE WS-WORK-AREAS
           
           OPEN I-O   PORTFOLIO-FILE
           OPEN INPUT UPDATE-FILE

      *-- Change: Open price feed file
           OPEN INPUT PRICEFEED-FILE

           IF NOT WS-SUCCESS-STATUS OR 
              NOT WS-UPDT-SUCCESS
              DISPLAY 'Error opening files: ' 
                      'PORT=' WS-FILE-STATUS
                      'UPDT=' WS-UPDT-STATUS
              MOVE WS-ERROR TO WS-RETURN-CODE
              PERFORM 3000-TERMINATE
           END-IF

      *-- Change: Check price feed file open status
           IF NOT WS-PRICEFEED-SUCCESS
              DISPLAY 'Error opening price feed file: ' WS-PRICEFEED-STATUS
              MOVE 'PF01' TO WS-FEED-ERROR-CODE
              MOVE 'Y' TO WS-FEED-ERROR
           END-IF
           .
           
       2000-PROCESS.
           READ UPDATE-FILE
               AT END
                   SET END-OF-FILE TO TRUE
               NOT AT END
                   PERFORM 2100-PROCESS-UPDATE
           END-READ
           .
           
       2100-PROCESS-UPDATE.
           MOVE UPDT-KEY TO PORT-KEY
           
           READ PORTFOLIO-FILE
           
           IF WS-SUCCESS-STATUS
      *-- Change: Integrate real-time price feed before applying update
               PERFORM 2150-APPLY-REALTIME-PRICE
               PERFORM 2200-APPLY-UPDATE
           ELSE
               ADD 1 TO WS-ERROR-COUNT
               DISPLAY 'Record not found: ' PORT-KEY
           END-IF
           .

      *-- Change: New paragraph for real-time price feed integration
       2150-APPLY-REALTIME-PRICE.
           MOVE 'N' TO WS-PRICE-FOUND
           MOVE 'N' TO WS-FEED-ERROR
           REWRITE-PRICEFEED.
               READ PRICEFEED-FILE
                   AT END
                       MOVE 'Y' TO WS-FEED-ERROR
                       MOVE 'PF02' TO WS-FEED-ERROR-CODE
                   NOT AT END
                       IF PF-PORTFOLIO-ID = PORT-ID
                           MOVE PF-PRICE TO PORT-LATEST-PRICE *-- Change: Store latest price in portfolio
                           MOVE PF-TIMESTAMP TO PORT-LATEST-PRICE-TIMESTAMP *-- Change: Store timestamp
                           MOVE 'Y' TO WS-PRICE-FOUND
                       END-IF
               END-READ
           IF WS-PRICE-FOUND = 'N'
               MOVE 'Y' TO WS-FEED-ERROR
               MOVE 'PF03' TO WS-FEED-ERROR-CODE
           END-IF
           .

       2200-APPLY-UPDATE.
           EVALUATE TRUE
               WHEN UPDT-STATUS
                   MOVE UPDT-NEW-VALUE TO PORT-STATUS
               WHEN UPDT-NAME
                   MOVE UPDT-NEW-VALUE TO PORT-CLIENT-NAME
               WHEN UPDT-VALUE
                   MOVE UPDT-NEW-VALUE TO WS-NUMERIC-WORK
                   MOVE WS-NUMERIC-WORK TO PORT-TOTAL-VALUE
           END-EVALUATE

      *-- Change: If price feed error, log audit (pseudo, actual logging handled elsewhere)
           IF WS-FEED-ERROR = 'Y'
               DISPLAY 'Price feed error for: ' PORT-KEY ' CODE: ' WS-FEED-ERROR-CODE *-- Change: Trace price feed error
           END-IF

           REWRITE PORT-RECORD
           
           IF WS-SUCCESS-STATUS
               ADD 1 TO WS-UPDATE-COUNT
           ELSE
               ADD 1 TO WS-ERROR-COUNT
               DISPLAY 'Update failed for: ' PORT-KEY
           END-IF
           .
           
       3000-TERMINATE.
           CLOSE PORTFOLIO-FILE
                 UPDATE-FILE

      *-- Change: Close price feed file
                 PRICEFEED-FILE
           
           DISPLAY 'Updates processed: ' WS-UPDATE-COUNT
           DISPLAY 'Errors occurred:  ' WS-ERROR-COUNT
           
           MOVE WS-RETURN-CODE TO RETURN-CODE
           .

[Summary Change Report]
- Added logic to read and apply real-time price feed for each portfolio update.
- Extended file control and working-storage for price feed integration and error handling.
- Modified 2000-PROCESS, 2100-PROCESS-UPDATE, and added 2150-APPLY-REALTIME-PRICE for feed ingestion and error trace.
- Inline comments added for traceability.
